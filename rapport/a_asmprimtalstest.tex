\chapter{Asm primtalstest}
\label{a:asmtestprogram}
Dette er primtalsprogrammet fra \kref{s:primtalstest} ``som en assembler programmør ville have skrevet programmet''.


\begin{footnotesize}\begin{verbatim}
DOSSEG
.MODEL SMALL
.STACK 200h

.DATA

.CODE

     mov ax,@DATA
     mov ds,ax

	mov cx, 31999
again:
	mov si, 2				; si = 2
primeTsTStart:
	cmp si, cx 				; while cx > si
	je primeTstEnd			;
		
	mov ax, cx		
	xor dx, dx				
	div si
	cmp dx, 0 				; if cx % si == 0 then stop testing current number
	je primeTstEnd:
	inc si					; si++
	jmp primeTstStart
primeTstEnd:
	
	loop again				; if cx > 0

	; END OF SHOW
    mov ah,4ch
    int 21h

END
\end{verbatim}\end{footnotesize}
